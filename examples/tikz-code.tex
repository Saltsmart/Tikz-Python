    % fill intersections

    \begin{scope}

    % intersection of first and third (1,3)

    \end{scope}
    % intersection of first and second (1,2)
    \begin{scope}
        \clip \boundingbox \thirdellipse;

        \fill[gray] \secondellipse;
    \end{scope}

    % (1,4)
    \begin{scope}
        \clip \boundingbox \thirdellipse;
        \clip \boundingbox \secondellipse;

    \end{scope}

        \clip \boundingbox \thirdellipse;
